To switch from Firebase to a Google Spreadsheet backend, we need to create a **bridge**. Google Sheets cannot be accessed directly from a browser for "writing" without a middleman. We will use **Google Apps Script** as a "Web App" that acts as your private API. ### Part 1: The Google Spreadsheet Setup 1. Create a new Google Spreadsheet. 2. Create **5 Sheets** inside that spreadsheet with these exact names and headers (case sensitive): * **Tabs**: `id`, `name`, `order`, `bgColor`, `bgImage`, `subject`, `year` * **Announcements**: `id`, `tabId`, `title`, `content`, `category`, `pinned`, `order`, `subject`, `year` * **Reminders**: `id`, `title`, `description`, `date`, `type`, `category`, `location`, `link`, `order`, `subject`, `year` * **Classes**: `id`, `name`, `subject`, `year` * **Collections**: `id`, `tabId`, `title`, `order`, `pinned`, `postIds`, `subject`, `year` --- ### Part 2: The Google Apps Script (The Backend) 1. In your Spreadsheet, go to **Extensions > Apps Script**. 2. Delete everything in `Code.gs` and paste the following: ```javascript /** * GOOGLE APPS SCRIPT BACKEND FOR GDCHUB * Allows GET (Fetch) and POST (Save/Delete) operations */ const SS = SpreadsheetApp.getActiveSpreadsheet(); function doGet(e) { const subject = e.parameter.subject; const year = e.parameter.year; const data = { tabs: getSheetData("Tabs", subject, year), announcements: getSheetData("Announcements", subject, year), reminders: getSheetData("Reminders", subject, year), classes: getSheetData("Classes", subject, year), collections: getSheetData("Collections", subject, year) }; return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const params = JSON.parse(e.postData.contents); const action = params.action; // 'save' or 'delete' const type = params.type; // 'Tabs', 'Announcements', etc. const item = params.data; const sheet = SS.getSheetByName(type); const headers = sheet.getRange(1, 1, 1, sheet.getLastColumn()).getValues()[0]; if (action === 'save') { const data = sheet.getDataRange().getValues(); let rowIndex = -1; // Check if item exists for (let i = 1; i < data.length; i++) { if (data[i][0] == item.id) { rowIndex = i + 1; break; } } const rowValues = headers.map(h => item[h] !== undefined ? (Array.isArray(item[h]) ? JSON.stringify(item[h]) : item[h]) : ""); if (rowIndex > 0) { sheet.getRange(rowIndex, 1, 1, headers.length).setValues([rowValues]); } else { sheet.appendRow(rowValues); } } else if (action === 'delete') { const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (data[i][0] == item.id) { sheet.deleteRow(i + 1); break; } } } return ContentService.createTextOutput(JSON.stringify({status: "success"})) .setMimeType(ContentService.MimeType.JSON); } function getSheetData(sheetName, subject, year) { const sheet = SS.getSheetByName(sheetName); const data = sheet.getDataRange().getValues(); const headers = data.shift(); return data .map(row => { let obj = {}; headers.forEach((h, i) => { let val = row[i]; // Parse JSON strings back to arrays (for postIds) if (typeof val === 'string' && (val.startsWith('[') || val.startsWith('{'))) { try { val = JSON.parse(val); } catch(e) {} } obj[h] = val; }); return obj; }) .filter(item => item.subject == subject && item.year == year); } ``` 3. **Click Deploy > New Deployment**. 4.
